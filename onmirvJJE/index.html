题目

给你一个正整数 n ，生成一个包含 1 到 n^2 所有元素，且元素按顺时针顺序螺旋排列的 n x n 正方形矩阵 matrix 。

示例 1：

输入：n = 3
输出：[[1,2,3],[8,9,4],[7,6,5]]

示例 2：

输入：n = 1
输出：[[1]]

提示：

  • 1 <= n <= 20

思路

差不多是最省脑子的一类题了，直接一个状态机即可，注意好边界控制，小心阴沟里翻船

直接上代码，细节在注释了~

代码

func generateMatrix(n int) [][]int {
	// 定义四个边界, 初始化结果矩阵
	left, right, top, bottom := 0, n-1, 0, n-1
	matrix := make([][]int, n)
	for i := range matrix {
		matrix[i] = make([]int, n)
	}

	num, maxNum := 1, n*n
	for num <= maxNum {
		// 最上面一排，从左往右遍历
		for i := left; i <= right; i++ {
			matrix[top][i] = num
			num++
		}
		// 最上面一排扫描完，上边界需要调整
		top++
		// 最右边一列，从上往下遍历
		for i := top; i <= bottom; i++ {
			matrix[i][right] = num
			num++
		}
		// 最右边一列扫描完，右边界需要调整
		right--
		// 最下边一排，从右往左遍历
		for i := right; i >= left; i-- {
			matrix[bottom][i] = num
			num++
		}
		// 最下边一排扫描完，下边界需要调整
		bottom--
		// 最左边一列，从下往上遍历
		for i := bottom; i >= top; i-- {
			matrix[i][left] = num
			num++
		}
		// 最左边一列扫描完，左边界需要调整
		left++
	}
	return matrix
}
